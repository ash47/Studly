//LOGIN AND PROFILE

/api/session
    +GET, Get User Information for currently logged in user!
        return: {
            username: STRING,
            email: STRING,
            success: true
        }
    +POST, Login with a user

        args: {
            username: String,
            password: String
        }
    +DELETE logout
        -Logout
        -Error, 404 Session Not Found
        -Success, 200 Logged Out
